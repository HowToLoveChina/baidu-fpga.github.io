Requirements

Alignment

FPGA matrix library designers should pay attention to these alignment requirements listed below.

Int8GEMM

Int8 GEMM operation should have following minimal size:

As for [m, k] x [k, n] => [m, n]

  • Data Matrix: [8 128]
  • Weight Matrix: [128 128]
  • Result Matrix: [128 128]

Int8CONV

Int8 convolution operation should have following alignment requirements:

For input image:

  • Image Batch: align to 8
  • Width * Channels: align to 32

For convolution kernels:

  • Kernel Num: align to 64
  • Window Size: align to 16

Note:

  • Window Size = (kern_h * kern_w * kern_c).
  • Kernel Num is the same as output channels of convolution.

DNN Example

int test_dnn(const size_t m,
const size_t n,
const size_t k) {
const size_t size_a = fpga::INT8_GEMM_DATA_SIZE(m, k);
const size_t size_b = fpga::INT8_GEMM_WEIGHT_SIZE(k, n);
const size_t size_c = fpga::INT8_GEMM_DATA_SIZE(m, n);
auto a = AlloateCPU<float>(m * k);
auto b = AlloateCPU<float>(k * n);
auto b_int8 = AlloateCPU<int8_t >(k * n);
auto c = AlloateCPU<float>(m * n);
auto c_cpu = AlloateCPU<float>(m * n);
auto a_encoded = AlloateCPU<float>(size_a);
auto b_encoded = AlloateCPU<int8_t>(size_b);
auto c_encoded = AlloateCPU<float>(size_c);
auto a_fpga = AlloateFPGA<float>(size_a);
auto b_fpga = AlloateFPGA<int8_t>(size_b);
auto c_fpga = AlloateFPGA<float>(size_c);
const float max_a = fill_float_randomly(a, m * k, -1.f, 1.f);
const float max_b = fill_float_randomly(b, k * n, -1.f, 1.f);
Assert(fpga::int8_gemm_align_data(m, k, a_encoded, a) == 0, "");
Assert(fpga::int8_gemm_encode_weights(m, k, b_encoded, b) == 0, "");
Assert(fpga::copy_to_fpga(dev, a_fpga, a_encoded, size_a) == 0, "");
Assert(fpga::copy_to_fpga(dev, b_fpga, b_encoded, size_b) == 0, "");
float scale = quantize(b_int8, b, k * n);
Assert(fpga::cpu_int8_gemm(m, n, k, a, b_int8, c_cpu, &scale, NULL, 0) == 0, "");
Assert(fpga::int8_gemm(dev, m, n, k, 1.0f,
a_fpga, b_fpga, 0.f, c_fpga, NULL, 0) == 0, "");
Assert(fpga::copy_from_fpga(dev, c_encoded, c_fpga, size_c) == 0, "");
Assert(fpga::int8_gemm_unalign_data(m, n, c, c_encoded) == 0, "");
Assert(compare_float_array(c, c_cpu, m * n), "Result error.");
free(a);
free(b);
free(c);
free(c_cpu);
free(a_encoded);
free(b_encoded);
free(c_encoded);
Assert(fpga::free(dev, a_fpga) == 0, "");
Assert(fpga::free(dev, b_fpga) == 0, "");
Assert(fpga::free(dev, c_fpga) == 0, "");
return 0;
}
